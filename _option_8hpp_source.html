760  std::string get_name(bool positional = false,
761  bool all_options = false
762  ) const {
763  if(get_group().empty())
764  return {}; // Hidden
765 
766  if(all_options) {
767 
768  std::vector<std::string> name_list;
769 
771  if((positional && (!pname_.empty())) || (snames_.empty() && lnames_.empty())) {
772  name_list.push_back(pname_);
773  }
774  if((get_items_expected() == 0) && (!fnames_.empty())) {
775  for(const std::string &sname : snames_) {
776  name_list.push_back("-" + sname);
777  if(check_fname(sname)) {
778  name_list.back() += "{" + get_flag_value(sname, "") + "}";
779  }
780  }
781 
782  for(const std::string &lname : lnames_) {
783  name_list.push_back("--" + lname);
784  if(check_fname(lname)) {
785  name_list.back() += "{" + get_flag_value(lname, "") + "}";
786  }
787  }
788  } else {
789  for(const std::string &sname : snames_)
790  name_list.push_back("-" + sname);
791 
792  for(const std::string &lname : lnames_)
793  name_list.push_back("--" + lname);
794  }
795 
796  return detail::join(name_list);
797  }
798 
799  // This returns the positional name no matter what
800  if(positional)
801  return pname_;
802 
803  // Prefer long name
804  if(!lnames_.empty())
805  return std::string(2, '-') + lnames_[0];
806 
807  // Or short name if no long name
808  if(!snames_.empty())
809  return std::string(1, '-') + snames_[0];
810 
811  // If positional is the only name, it's okay to use that
812  return pname_;
813  }

820  void run_callback() {
821 
822  if(current_option_state_ == option_state::parsing) {
823  _validate_results(results_);
824  current_option_state_ = option_state::validated;
825  }
826 
827  if(current_option_state_ < option_state::reduced) {
828  _reduce_results(proc_results_, results_);
829  current_option_state_ = option_state::reduced;
830  }
831  if(current_option_state_ >= option_state::reduced) {
832  current_option_state_ = option_state::callback_run;
833  if(!(callback_)) {
834  return;
835  }
836  const results_t &send_results = proc_results_.empty() ? results_ : proc_results_;
837  bool local_result = callback_(send_results);
838 
839  if(!local_result)
840  throw ConversionError(get_name(), results_);
841  }
842  }

845  const std::string &matching_name(const Option &other) const {
846  static const std::string estring;
847  for(const std::string &sname : snames_)
848  if(other.check_sname(sname))
849  return sname;
850  for(const std::string &lname : lnames_)
851  if(other.check_lname(lname))
852  return lname;
853 
854  if(ignore_case_ ||
855  ignore_underscore_) { // We need to do the inverse, in case we are ignore_case or ignore underscore
856  for(const std::string &sname : other.snames_)
857  if(check_sname(sname))
858  return sname;
859  for(const std::string &lname : other.lnames_)
860  if(check_lname(lname))
861  return lname;
862  }
863  return estring;
864  }

869  bool check_name(const std::string &name) const {
870 
871  if(name.length() > 2 && name[0] == '-' && name[1] == '-')
872  return check_lname(name.substr(2));
873  if(name.length() > 1 && name.front() == '-')
874  return check_sname(name.substr(1));
875  if(!pname_.empty()) {
876  std::string local_pname = pname_;
877  std::string local_name = name;
878  if(ignore_underscore_) {
879  local_pname = detail::remove_underscore(local_pname);
880  local_name = detail::remove_underscore(local_name);
881  }
882  if(ignore_case_) {
883  local_pname = detail::to_lower(local_pname);
884  local_name = detail::to_lower(local_name);
885  }
886  if(local_name == local_pname) {
887  return true;
888  }
889  }
890 
891  if(!envname_.empty()) {
892  // this needs to be the original since envname_ shouldn't match on case insensitivity
893  return (name == envname_);
894  }
895  return false;
896  }

918  std::string get_flag_value(const std::string &name, std::string input_value) const {

1173  void _validate_results(results_t &res) const {
1174  // Run the Validators (can change the string)
1175  if(!validators_.empty()) {
1176  if(type_size_max_ > 1) { // in this context index refers to the index in the type
1177  int index = 0;

1179  multi_option_policy_ == CLI::MultiOptionPolicy::TakeLast) {
1180  // create a negative index for the earliest ones
1181  index = get_items_expected_max() - static_cast<int>(res.size());
1182  }

1190  if(!err_msg.empty())

1194  } else {

1214  void _reduce_results(results_t &res, const results_t &original) const {
1215 
1216  // max num items expected or length of vector, always at least 1
1217  // Only valid for a trimming policy
1218 
1219  res.clear();
1220  // Operation depends on the policy setting
1221  switch(multi_option_policy_) {
1222  case MultiOptionPolicy::TakeAll:
1223  break;
1224  case MultiOptionPolicy::TakeLast: {
1225  // Allow multi-option sizes (including 0)
1226  std::size_t trim_size = std::min<std::size_t>(
1227  static_cast<std::size_t>(std::max<int>(get_items_expected_max(), 1)), original.size());
1228  if(original.size() != trim_size) {
1229  res.assign(original.end() - static_cast<results_t::difference_type>(trim_size), original.end());
1230  }
1231  } break;
1232  case MultiOptionPolicy::TakeFirst: {
1233  std::size_t trim_size = std::min<std::size_t>(
1234  static_cast<std::size_t>(std::max<int>(get_items_expected_max(), 1)), original.size());
1235  if(original.size() != trim_size) {
1236  res.assign(original.begin(), original.begin() + static_cast<results_t::difference_type>(trim_size));
1237  }
1238  } break;
1239  case MultiOptionPolicy::Join:
1240  if(results_.size() > 1) {
1241  res.push_back(detail::join(original, std::string(1, (delimiter_ == '\0') ? '\n' : delimiter_)));
1242  }
1243  break;
1244  case MultiOptionPolicy::Throw:
1245  default: {
1246  auto num_min = static_cast<std::size_t>(get_items_expected_min());
1247  auto num_max = static_cast<std::size_t>(get_items_expected_max());
1248  if(num_min == 0) {
1249  num_min = 1;
1250  }
1251  if(num_max == 0) {
1252  num_max = 1;
1253  }
1254  if(original.size() < num_min) {
1255  throw ArgumentMismatch::AtLeast(get_name(), static_cast<int>(num_min), original.size());
1256  }
1257  if(original.size() > num_max) {
1258  throw ArgumentMismatch::AtMost(get_name(), static_cast<int>(num_max), original.size());
1259  }
1260  break;
1261  }
1262  }
1263  }
